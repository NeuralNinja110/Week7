import pandas as pd
import matplotlib.pyplot as plt

# Employee data as CSV string
data = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,HR,Africa,73.03,6,3.2
EMP002,HR,Africa,70.48,1,3.1
EMP003,HR,Asia Pacific,76.36,14,3.5
EMP004,Finance,Asia Pacific,89.31,4,4.1
EMP005,IT,Latin America,69.42,7,4.5
"""

from io import StringIO
df = pd.read_csv(StringIO(data))

# Calculate frequency count for "Operations" department
operations_count = (df['department'] == 'Operations').sum()
print("Frequency count for Operations department:", operations_count)

# Create histogram of department distribution
plt.figure(figsize=(6, 4))
df['department'].value_counts().plot(kind='bar')
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Count")
plt.show()
